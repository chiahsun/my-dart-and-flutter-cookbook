[source,dart]
----
import 'package:bloc/bloc.dart';
import 'package:meta/meta.dart';

part 'counter_state.dart';

class CounterCubit extends Cubit<CounterState> {
  CounterCubit() : super(CounterInitial());
}
----

`counter_state.dart`

[source,dart]
----

sealed class CounterState {}
